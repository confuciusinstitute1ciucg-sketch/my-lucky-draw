<div class="container">
    <h2>🎯 实时抽签系统 (1-6号)</h2>
    <div class="form">
        <input type="text" id="stuName" placeholder="输入姓名（抽后不可更改）">
        <button onclick="draw()">开始抽签</button>
    </div>
    <div class="grid" id="grid"></div>
    <div class="admin"><button style="background:#ff4d4f" onclick="resetPool()">教师重置</button></div>
</div>
<script>
    async function refresh() {
        const res = await fetch('/api/status');
        const slots = await res.json();
        document.getElementById('grid').innerHTML = slots.map(s => `
            <div class="card ${s.isTaken?'taken':''}">
                <h3>${s.id} 号</h3>
                <div>${s.isTaken ? s.studentName : '🔍 待抽取'}</div>
            </div>
        `).join('');
    }
    async function draw() {
        const name = document.getElementById('stuName').value.trim();
        if(!name) return alert('请输入名字！');
        const res = await fetch('/api/draw', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ name }) });
        const data = await res.json();
        alert(data.msg || `成功抽到 ${data.num} 号！`);
        if(data.success) document.getElementById('stuName').value = '';
        refresh();
    }
    async function resetPool() {
        const password = prompt('请输入重置密码:');
        if(!password) return;
        const res = await fetch('/api/reset', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ password }) });
        const data = await res.json();
        alert(data.msg);
        refresh();
    }
    refresh();
    setInterval(refresh, 2000); // ✨ 每隔2秒自动更新数据，实现实时同步
</script>
